CREATE TABLE `stock_money_flow` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '主键',
  `isvalid` tinyint(2) DEFAULT '1' COMMENT '状态--0：有效，1：失效',
  `version` int(11) DEFAULT '1' COMMENT '乐观锁，解决并发访问',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '数据创建时间',
  `create_user` varchar(50) DEFAULT NULL COMMENT '数据创建人',
  `update_time` datetime DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP COMMENT '数据修改时间',
  `update_user` varchar(50) DEFAULT NULL COMMENT '数据修改人',
  `deal_time` varchar(20) DEFAULT NULL COMMENT '时间',
  `stock_code` varchar(10) DEFAULT NULL COMMENT '编码',
  `magor_money` varchar(20) DEFAULT '0' COMMENT '主力资金净流入',
  `super_money` varchar(20) DEFAULT '0' COMMENT '超大单资金净流入',
  `big_money` varchar(20) DEFAULT '0' COMMENT '大单资金净流入',
  `middle_money` varchar(20) DEFAULT '0' COMMENT '中单资金净流入',
  `small_money` varchar(20) DEFAULT '0' COMMENT '小单资金净流入',
  PRIMARY KEY (`id`) USING BTREE,
  KEY `deal_time` (`deal_time`),
  KEY `stock_code` (`stock_code`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8mb4 COMMENT='每日资金流量表'
