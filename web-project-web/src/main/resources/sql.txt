CREATE TABLE `stock_tiger_list` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '主键',
  `isvalid` tinyint(2) DEFAULT '1' COMMENT '状态--0：有效，1：失效',
  `version` int(11) DEFAULT '1' COMMENT '乐观锁，解决并发访问',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '数据创建时间',
  `create_user` varchar(50) DEFAULT NULL COMMENT '数据创建人',
  `update_time` datetime DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP COMMENT '数据修改时间',
  `update_user` varchar(50) DEFAULT NULL COMMENT '数据修改人',
  `stock_code` varchar(20) DEFAULT NULL COMMENT '股票编码',
  `stock_name` varchar(20) DEFAULT NULL COMMENT '股票名称',
  `publish_time` varchar(20) DEFAULT NULL COMMENT '发布日期',
  `upper_range` decimal(10,2) DEFAULT '0.00' COMMENT '涨跌幅',
  `hand_rate` decimal(10,2) DEFAULT '0.00' COMMENT '换手率',
  `focus_reason` varchar(200) DEFAULT NULL COMMENT '上榜原因',
  `buy_money` decimal(20,2) DEFAULT '0.00' COMMENT '买入额',
  `sell_money` decimal(20,2) DEFAULT '0.00' COMMENT '卖出额',
  `unusual_reason` varchar(2000) DEFAULT NULL COMMENT '异动原因',
  `choose_reason` varchar(2000) DEFAULT NULL COMMENT '选择原因',
  PRIMARY KEY (`id`) USING BTREE
) ENGINE=InnoDB AUTO_INCREMENT=146 DEFAULT CHARSET=utf8mb4 COMMENT='龙虎榜'
