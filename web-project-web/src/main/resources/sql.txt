CREATE TABLE `english_root_affix` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '主键',
  `same_root` varchar(100) DEFAULT NULL COMMENT '相同词根',
  `root_affix` varchar(20) DEFAULT NULL COMMENT '词根词缀',
  `root_type` varchar(20) DEFAULT NULL COMMENT '类型',
  `root_mean` varchar(100) DEFAULT NULL COMMENT '示意',
  `root_resource` varchar(5000) DEFAULT NULL COMMENT '词根来源',
  PRIMARY KEY (`id`) USING BTREE
) ENGINE=InnoDB AUTO_INCREMENT=40 DEFAULT CHARSET=utf8mb4 COMMENT='词根词缀表'
